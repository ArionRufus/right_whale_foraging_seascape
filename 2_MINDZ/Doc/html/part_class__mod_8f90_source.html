606 ! Daytime preferred depth
607 real function zday_funz(this, surf) result(zval)
608 class(zoo) :: this
609 real, optional :: surf
610
611 if ( present(surf) ) then
612 zval = 7.53 * max(24.,surf) - 64.2 ! M. norvegica
613 else
614 zval = 10.
615 endif
616
617 end function zday_funz
618
619 ! Nighttime preferred depth
620 real function znight_funz(this) result(zval)
621 class(zoo) :: this
622 integer :: k
623 real :: r1
624
625 call rng_uni(r1)
626 k = max( 1, ceiling( r1 * 10. ) )
627 zval = mid_depth(k)
628
629 end function znight_funz
