407 ! Daytime preferred depth
408 real function zday_funk(this, surf) result(zval)
409 class(krill) :: this
410 real, optional :: surf
411
412 if ( present(surf) ) then
413 zval = 11.43 * max(24.,surf) - 200.87 ! T. raschii
414 else
415 zval = 10.
416 endif
417
418 end function zday_funk

420 ! Nightime preferred depth
421 real function znight_funk(this) result(zval)
422 class(krill) :: this
423
424 call rng_uni(r1)
425 k = max( 1, ceiling( r1 * 10. ) )
426 zval = mid_depth(k)
427
428 end function znight_funk

430 ! Arrhenius function of temperature = Universal Temperature Dependance
431 ! (Gillooly et al. 2001)
432 ! Returns a temperature dependance coefficient (unitless)
433 real function arrheniusk(this, e_act)
434 class(krill) :: this
435 real :: e_act
436 real, parameter :: k = 8.62e-5 ! The Boltzmann's constant in eV.K-1
437 real, parameter :: t0 = 273.15 ! Reference temperature (the frozing point of water) in K
438
439 arrheniusk = exp( ( e_act * this%tp ) / ( k * ( this%tp + t0 ) * t0 ) )
440
441 end function arrheniusk
442
443 ! Arrhenius adjustement function ensures an exponential decrease
444 ! AFTER some threshold
445 real function adj_arrk(this, alpha_temp)
446 class(krill) :: this
447 real :: alpha_temp, dtemp
448
449 dtemp = this%t_lim - this%tp
450
451 if( dtemp > 0 ) then
452 adj_arrk = 1
453 else
454 adj_arrk = exp( dtemp * alpha_temp )
455 endif
456
457 end function adj_arrk

459 ! Development time (intermoult period) in days
460 real function devlpk(this)
461 class(krill) :: this
462
463 devlpk = 86400. * ( this%a_molt + this%b_molt * this%tp )
464
465 if ( devlpk < 3. * 86400. ) devlpk = 3. * 86400.
466
467 end function devlpk
468
469 ! Development freq of the individual.
470 ! when the freq > 0.4, computes the molt_factor which fixes
471 ! the next individual's length
472 ! When the freq > 1, updates the individual's length with the
473 ! previously computed molt_factor.
474 subroutine moltk(this)
475 class(krill) :: this
476
477 if ( this%cell_out .eqv. .true. ) then
478 this%dev_frac = 0.0
479 else
480
481 this%dev_frac = this%dev_frac + ( dt / this%devlp() )
482
483 ! NOTE: moulting is irreversibly engaged after 40% of the IMP.
484 ! The new individuals length is completely decided at this time
485 ! according to the allometric relationship
486 ! if ( this%dev_frac > 0.4 .and. this%molt_length < EPS ) then
487 ! this%molt_length = ( this%mass / this%aw )** ( 1 / this%bw )
488 ! endif
489
490 ! At the end of the development phase, the individual molts
491 if ( this%dev_frac > 1. ) then
492 this%mass = this%mass
493 this%dev_frac = this%dev_frac - 1.
494 endif
495
496 endif
497
498 end subroutine moltk
499
500 ! Respiration = basal metabolism + activity (swimming) metabolism
501 ! but NOT specific dynamic action as it is included in assimilation
502 ! coeffcient
503 ! Data from Angelique
504 real function respk(this)
505 class(krill) :: this
506 real, parameter :: alpha = 0.2
507
508 respk = this%r0 &
509 * this%mass ** this%rb &
510 * this%arrhenius(this%er) &
511 * this%adj_arr(alpha) &
512 * dt
513
514 end function respk
515
516 ! Updates the new mass of the individual
517 subroutine growk(this)
518 class(krill) :: this
519
520 if ( this%cell_out .eqv. .true. ) then
521 this%mass = this%mass
522 else
523 this%mass = this%mass - this%resp() !+ this%A * this%ingest()
524 endif
525
526 end subroutine growk

1080
1081 subroutine predation (this, sss, z_depth, par_surf, temps, ph)
1082 class(calanus) :: this
1083 real, intent(in):: sss,z_depth, par_surf, temps, ph
1084 real,parameter :: factor_n2c=79.5729
1085 real,parameter :: ratio_c2chl=0.0180
1086 real,parameter :: iomax=360
1087 real,parameter :: min1=0, max1=iomax, min2=0.1, max2=0.9, kw=0.04
1088 real :: lin_trans_term1,phytot, phyt_surf
1089 real :: io, kcdom, kchla, iv, iprime
1090 real,parameter :: k_exp=0.002, k_prime=0.15
1091 real :: kmort, mort_d, mort_nonvis_daily_max, mort_nonvis_daily_min, mort_asym
1092 real :: ino, mort_vis_daily, mort_vis_dt
1093 real :: minn1, minn2, lin_trans_term2, maxx1, maxx2
1094 real, parameter :: tmin=0.0
1095 real :: mort_nonvis, mort_nonvis_dt, tmp
1096
1097 ! visual predation
1098
1099 !get surface chla at (x,y)
1100 lin_trans_term1=(max2-min2)/(max1-min1)
1101 phytot=ph*factor_n2c
1102 phyt_surf=phytot*ratio_c2chl
1103
1104 ! calculate light attenuation kd
1105 io=par_surf
1106 kcdom=-0.0364*sss+1.1942
1107 kchla=0.0518*phyt_surf**(0.428)
1108 kd=kw+kcdom+kchla
1109
1110
1111 ! calculate light dependent mortality
1112
1113 iv = iomax*exp(-kd*z_depth)
1114 iprime=lin_trans_term1*(iv-max1)+max2
1115
1116
1117 kmort=k_prime**4.0
1118 mort_d=kmort*24.0
1119 mort_asym=mort_d*2.5
1120 mort_nonvis_daily_max=0.1*mort_asym
1121 mort_nonvis_daily_min=0.01*mort_asym
1122
1123 mort_vis_daily = iprime*mort_asym*(1.0-exp(-k_exp*this%mass))
1124 mort_vis_dt=mort_vis_daily/48.0
1125
1126 ! non_visual_predation
1127
1128 minn1=tmin
1129 maxx1=18.59 ! the number was calculated in matlab, gulf region, 20180101 to 20190101, top 4 layer averaged
1130 minn2=mort_nonvis_daily_min
1131 maxx2=mort_nonvis_daily_max
1132 lin_trans_term2=(maxx2-minn2)/(maxx1-minn1)
1133
1134 tmp=max(0.0, temps)
1135 mort_nonvis=lin_trans_term2*(tmp-maxx1)+maxx2
1136 mort_nonvis_dt=mort_nonvis/48.0
1137
1138 ! calculate the abundance change due to visual and non visual predation
1139 this%abund=this%abund*(1.0-mort_vis_dt-mort_nonvis_dt)

1175 endif

1190 endif
