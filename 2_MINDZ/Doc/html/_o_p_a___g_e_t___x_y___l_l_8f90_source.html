171 SUBROUTINE opa_proj_inv

173 IMPLICIT NONE

175 REAL :: lonbl,latbl,lonbr,latbr,lontl,lattl,lontr,lattr

177 REAL :: r1,r2,dx,dy,dx1,dy1,dri00,drj00
178
179 if (first) then
180 call opa_proj_init
181 first = .false.
182 end if
183
184 ! Find nearest wet i,j to lon00,lat00
185 i00 = min(ni,max(1,int(ri00)))
186 j00 = min(nj,max(1,int(rj00)))
187
188 dri00 = ri00-1.0*(i00)
189 drj00 = rj00-1.0*(j00)
190
191 lonbl = nav_lon(i00,j00)
192 latbl = nav_lat(i00,j00)
193 lontl = nav_lon(i00,min(nj,j00+1))
194 lattl = nav_lat(i00,min(nj,j00+1))
195 lonbr = nav_lon(min(ni,i00+1),j00)
196 latbr = nav_lat(min(ni,i00+1),j00)
197 lontr = nav_lon(min(ni,i00+1),min(nj,j00+1))
198 lattr = nav_lat(min(ni,i00+1),min(nj,j00+1))
199
200 lon00 = lonbl*(1-dri00)*(1-drj00) + lontl*(1-dri00)*drj00 + &
201 lonbr*dri00*(1-drj00) + lontr*dri00*drj00
202
203 lat00 = latbl*(1-dri00)*(1-drj00) + lattl*(1-dri00)*drj00 + &
204 latbr*dri00*(1-drj00) + lattr*dri00*drj00
205
206 return

321 REAL function geproj_dist(lon,lat,lon0,lat0)
322
323 ! % result is in meters
324 ! % REARTH = 6.3675E6;
325 ! % REARTH = 6367500; %Webdrgoue???
326
327 REAL :: lon,lat,lon0,lat0
328
329 REAL :: rearth,pi
330
331 parameter(rearth = 6378137.0,pi = 3.1415)! m_ll2dist
332
333 REAL :: dlat,dlon,sdlat,sdlon,ee,cc,clat0
334
335 ! % this is an approximation that assumes region is flat enough for
336 ! % pathagoras to work. Use Haversine formula
337 ! %y = (lat-lat0)*REARTH*pi/180;
338 ! %x = (lon-lon0)*REARTH*(pi/180) .* cos(lat*pi/180);
339 !Haversine formula
340 dlat = (lat-lat0)*pi/180;
341 dlon = (lon-lon0)*pi/180 ;
342 sdlat = sin(dlat/2)
343 sdlon = sin(dlon/2)
344 clat0 = cos(lat0 *pi/180 )
345 ee = sdlat*sdlat + clat0*clat0*sdlon*sdlon
346 cc = 2 * atan2(sqrt(ee),sqrt(1-ee));
347 geproj_dist = rearth * cc;
348
349 END FUNCTION geproj_dist
