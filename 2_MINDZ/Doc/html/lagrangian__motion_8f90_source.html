36 real function ou(theta, mu, sigma, dt, var)
37
38 ! OU(theta,mean,sigma,dt,var) returns a value following an Ornstein-Uhlenbeck
39 ! stochastic process.
40 !
41 ! Input:
42 ! theta(>0) : rate of mean reversion ~ swimming speed [time^-1]
43 ! mu : long-term mean of the process [unit]
44 ! sigma(>0) : volatility = average magnitude of the random
45 ! fluctuations modelled as Brownian motions [time^-0.5]
46 ! dt : time-step [time]
47 ! var : value of the variable at time t [unit]
48 !
49 ! Output:
50 ! var : value of the variable updated at t+dt [unit]
51 !
52 ! http://planetmath.org/encyclopedia/OrnsteinUhlenbeckProcess.html
53 !
54 ! MapsF 2011
55
56 integer, intent(in) :: dt
57 real, intent(in) :: theta, mu, sigma, var
58 real :: mean, sd, rn
59
60
61 mean = mu + (var-mu) * exp(-theta*dt)
62
63 sd = sigma * sqrt( (1-exp(-2*theta*dt)) / 2 / theta )
64
65 call rng_norm(rn)
66 ou = rn * sd + mean
67
68 return
69
70 end function ou

74 real function vert_mig(z, zday, znight, bottom, theta, sigma, dvm)
75
76 ! Return depth of particle according to a diel cycle
77 ! Includes a random component (Ornstein-Uhlenbeck distribution)
78 !
79 ! Input:
80 ! z = depth of the particule at the previous timestep
81 ! zday = target depth of the particule during daytime
82 ! znight = " " nighttime
83 ! bottom = bottom depth
84 ! theta = vertical "swimming" speed coefficient
85 ! sigma = vertical spread coefficient
86 ! dvm = particle migrate (.true.) or not (.false.)
87 !
88 ! Output:
89 ! vert_mig = vertical position of the particle
90 !
91 ! MapsF 2011
92
93
94 real, intent(in) :: z, zday, znight, bottom
95 real, intent(in) :: theta, sigma
96 logical, intent(in) :: dvm
97
98 real :: sig, zopt
99
100
101 zopt = min(bottom, zday)
102
103 ! Nighttime
104 if (.not.light_day .and. dvm) then
105 ! Forage close to the surface
106 zopt = min(bottom, znight)
107 endif
108
109 ! sigma (vertical spread) is function of zopt - bottom
110 ! 0.2 < f(zopt-bottom) < 1.0
111 ! with f(<0) = 1
112 ! f(50) = 1/2
113 ! f(100) = 1/3
114 ! ...
115 ! f(inf) = 1/5
116
117 sig = sigma * ( 0.2 + 50. / ( 50 + max(0.,zopt-bottom) ) ) / 1.2
118
119 ! no particles above the surface | below bottom
120
121 vert_mig = max(0.1, bottom - abs( bottom - abs( ou(theta,zopt,sig,dt,z) ) ) )
122
123 return
124
125 end function vert_mig

132 !
133 ! This routine interpolates linearly 3D array at a given point

178
179 kk = min(max(int(yp),1),nn-1)
180
181 dx = xp-(float(ii)-.5)
182
183 dy = yp-float(kk)
184
185 else

187 stop 'Index problems when calling value'
188
189 endif

191 update = 0.0

194
195 j1 = 1
196
197 j2 = 1
198
199 ! Find local vertical coordinate
200 i = min(max(int(xp)+1,1),nm)
201
202 k = min(max(int(yp)+1,1),nn)

204 if(nlayer(i,k)==0) then ! below the bottom or value = 0.
205 return
206
207 elseif(zp>dz(nlayer(i,k))) then
208
209 return
210
211 endif
212
213 if(index=='w') then ! Fields at the top of each layer(ex: w)
214
215 if(zp<=0.0) then ! If above the surface
216
217 d_z = 0.
218
219 j1 = 1
220
221 j2 = 1
222
223 else
224
225 loop1 : do j = 1,nlayer(i,k) ! Find the bracketing layers
226
227 if(zp>up_depth(j).and.zp<=dz(j)) then
228
229 d_z = (zp-up_depth(j))/(dz(j)-up_depth(j)) ! Thickness fraction
230
231 j1 = j
232
233 j2 = j+1
234
235 exit loop1 ! Exit the loop to save time
236
237 endif
238
239 enddo loop1
240
241 if(j1==nlayer(i,k)) then
242
243 j2 = j1 ! Constante value in the bottom layer
244
245 d_z = 0.
246
247 endif
248
249 endif
250
251 ! Fields at the center of each layer (ex:u)
252
253 elseif(index=='u'.or.index=='v'.or.index=='s') then
254
255 ! Constant value if above middle of top layer
256
257 if(zp<=mid_depth(1)) then
258
259 d_z = 0.
260
261 j1 = 1
262
263 j2 = 1
264
265 ! Constant value if if below the middle of bottom layer
266
267 elseif (nlayer(i,k)>0.and.zp>mid_depth(nlayer(i,k))) then
268
269 d_z = 0.
270
271 j1 = nlayer(i,k)
272
273 j2 = nlayer(i,k)
274
275 else
276
277 loop2 : do j = 1,nlayer(i,k)-1 ! Find the bracketing layers
278
279 if(zp>mid_depth(j).and.zp<=mid_depth(j+1)) then
280
281 thick = (dz(j)-up_depth(j))*0.5+(dz(j+1)-up_depth(j+1))*0.5
282
283 d_z = (zp-mid_depth(j))/thick ! Thickness fraction
284
285 j1 = j
286
287 j2 = j+1
288
289 exit loop2
290
291 endif
292
293 enddo loop2
294
295 endif
296
297 endif
298
299 !--- Horizontal interpolation (bilinear)
300
301 ! First horizontal plan
302
303 val1 = dx * dy * a3d(ii+1,kk+1,j1) + (1.-dx) * dy * a3d(ii,kk+1,j1) + &
304 dx * (1.-dy)* a3d(ii+1,kk,j1) + (1.-dx) * (1.-dy)* a3d(ii,kk,j1)
305
306 ! Second horizontal plan
307
308 val2 = dx * dy * a3d(ii+1,kk+1,j2) + (1.-dx) * dy * a3d(ii,kk+1,j2) + &
309 dx *(1.-dy)* a3d(ii+1,kk,j2) + (1.-dx) * (1.-dy)* a3d(ii,kk,j2)
310
311 ! Vertical interpolation (linear)
312
313 update = d_z*val2 + (1.-d_z)*val1
314
315 return
316
317 end function update
