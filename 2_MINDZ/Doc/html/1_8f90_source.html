180  ! devlp returns the develop time (intermoult period) in days
181 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
182  real function devlp(this)
183  class(krill) :: this
184 
185  devlp = 86400. * ( this%a_molt + this%b_molt * this%tp )
186 
187  if ( devlp < 3. * 86400. ) devlp = 3. * 86400.
188 
189  end function devlp

191 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
192  ! Arrhenius function of temperature : Universal Temperature Dependance (Gillooly et al. 2001)
193  ! Returns a temperature dependance coefficient (unitless)
194 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
195  real function arrhenius(this, e_act)
196  class(krill) :: this
197  real :: e_act
198  real, parameter :: k = 8.62e-5 ! The Boltzmann's constant in eV.K-1
199  real, parameter :: t0 = 273.15 ! Reference temperature (the frozing point of water) in K
200 
201  arrhenius = exp( ( e_act * this%tp ) / ( k * ( this%tp + t0 ) * t0 ) )
202 
203  end function arrhenius

205 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
206  ! Arrhenius adjustement function
207 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
208  real function adj_arr(this, alpha_temp)
209  class(krill) :: this
210  real :: alpha_temp, dtemp
211 
212  dtemp = this%t_lim - this%tp
213 
214  if( dtemp > 0 ) then
215  adj_arr = 1
216  else
217  adj_arr = exp( dtemp * alpha_temp )
218  endif
219 
220  end function adj_arr

222 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
223  ! resp returns the energy lost by respiration in mg C.s^-1
224  !
225  ! Respiration: basal metabolism + activity (swimming) metabolism
226  ! but NOT specific dynamic action as it is included in assimilation coeffcient
227  ! Data from Angelique
228 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
229  real function resp(this)
230  class(krill) :: this
231  real, parameter :: alpha = 0.2
232 
233  resp = this%r0 * this%mass**this%rb * this%arrhenius(this%er) * this%adj_arr(alpha) * this%dt
234 
235  end function resp

237 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
238  ! molt updates the development freq of the individual.
239  ! when the freq > 0.4, computes the molt_factor which fixes the next individual's length
240  ! When the freq > 1, updates the individual's length with the previously computed molt_factor.

257 ! if ( this%dev_freq > 0.4 .and. this%molt_length < EPS ) then
